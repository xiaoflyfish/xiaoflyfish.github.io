```javascript
class Solution {
    public ListNode getKthFromEnd(ListNode head, int k) {
        if (head == null) {
            return null;
        }
        //1->2->3->4->5
        ListNode pre = head, after = head;
        for (int i = 0; i < k;i++) {
            pre = pre.next;
        }
        while (pre != null) {
            pre = pre.next;
            after = after.next;
        }
        return after;
    }
}
```
